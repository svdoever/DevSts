Add the following setting to your web.config file: 

            <appSettings>
                <add key="ida:ADFSMetadata" value="http://DevSts/FederationMetadata" />
                <add key="ida:Wtrealm" value="urn:Whatever" />
                <add key="ida:ReplyAddress" value="<url of your site>" />
            </appSettings>
